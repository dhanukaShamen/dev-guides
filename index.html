// TypeScript 6 — strict mode by default
interface User {
  id: number;
  name: string;
  role: "admin" | "viewer";
}

function greet(user: User): string {
  const prefix =
    user.role === "admin"
      ? "Admin"
      : "Welcome";
  return `${prefix}, ${user.name}!`;
}

// TS catches this at compile time ✓
greet({ id: 1, name: "Alex", role: "admin" });
// React 19 — use() hook + Actions
import { use, useState } from "react";

interface Post {
  id: number; title: string;
}

function PostCard({ promise }: {
  promise: Promise<Post>
}) {
  // New in React 19 — use() unwraps promises
  const post = use(promise);
  return <h2>{post.title}</h2>;
}

Why Learn These Together?

Three technologies,
one complete stack.

TypeScript gives you type safety, React puts that safety into the UI layer, and Node.js + Express brings it to the server. Together they form the full JavaScript ecosystem — learn them as a trio and every concept reinforces the next.

TypeScript → Safer React Components
TS Interfaces document every prop a component accepts — no more runtime "undefined is not a function".
Node.js → The Runtime Under Everything
React's build tools, TypeScript's compiler, and Express all run on Node — understanding it unlocks the whole ecosystem.
Express APIs → Real React Data
The REST APIs you build with Express are exactly what React's use() hook and useEffect fetch from.
Async Patterns Carry Across All Three
Promises, async/await, and error handling learned in Node transfer directly to React hooks and TypeScript utility types.
Before You Start

What you'll need.

All three guides are beginner-friendly. Here's what to have ready before you open the first chapter.

Modern JavaScript (ES6+)

Comfort with arrow functions, destructuring, template literals, and async/await. All three guides build on these foundations — not past them.

Node.js 18+ & npm

Required for all three guides. Node runs TypeScript compilation, Express servers, and Vite's dev server. Install once — use everywhere.

VS Code

Recommended for all guides. Install the TypeScript and ESLint extensions to get the inline autocomplete and error highlighting the guides reference.

For TypeScript Guide

No prior TypeScript knowledge needed. A basic grasp of JavaScript objects and functions is enough to follow along from Chapter 1.

For React Guide

Uses Vite as the build tool — one scaffold command covered in Chapter 1. Basic HTML and JS knowledge is all you need to start.

For Node.js Guide

No backend experience required. If you can run a command in the terminal and understand basic JS functions, you're ready to build your first server.
